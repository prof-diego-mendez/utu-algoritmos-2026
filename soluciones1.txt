  resultado = (valor - 32) × 5/9
FIN SI

  resultado = celsius + 273.15

  resultado = valor - 273.15
FIN SI

  celsius = valor - 273.15
  resultado = celsius × 9/5 + 32
FIN SI

ESCRIBIR "Resultado: " + resultado
